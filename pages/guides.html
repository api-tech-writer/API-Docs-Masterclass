üöÄ Quick Start - 5 Minutes to First Request

Step 1: Sign Up for an API Key

Visit dashboard.globalweather.com to create your account and generate an API key.

Step 2: Set Up Your Environment

# Set your API key as an environment variable
export WEATHER_API_KEY="your_api_key_here"

# Or in .env file
echo "WEATHER_API_KEY=your_api_key_here" >> .env

Step 3: Make Your First Request

curl -X GET "https://api.globalweather.com/v1/weather/current?city=London" \
  -H "Authorization: Bearer $WEATHER_API_KEY"

Step 4: Parse the Response

{
  "location": {
    "city": "London",
    "country": "GB"
  },
  "current": {
    "temperature": 15.5,
    "weather": {
      "description": "partly cloudy"
    }
  }
}

Retry Strategy with Exponential Backoff

async function fetchWithRetry(url, options, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            const response = await fetch(url, options);
            
            if (response.ok) {
                return response;
            }
            
            if (response.status === 429 || response.status >= 500) {
                // Exponential backoff: 1s, 2s, 4s
                const delay = Math.pow(2, i) * 1000;
                await new Promise(r => setTimeout(r, delay));
                continue;
            }
            
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            
        } catch (error) {
            if (i === maxRetries - 1) throw error;
        }
    }
}

Step 1: Create Webhook Endpoint

# Flask webhook endpoint
from flask import Flask, request, jsonify
import hmac
import hashlib

app = Flask(__name__)
WEBHOOK_SECRET = 'your_webhook_secret'

@app.route('/webhooks/weather', methods=['POST'])
def handle_webhook():
    # Verify signature
    signature = request.headers.get('X-Webhook-Signature')
    if not verify_signature(request.data, signature):
        return jsonify({'error': 'Invalid signature'}), 401
    
    # Process event
    event = request.json
    
    if event['event'] == 'weather.alert.created':
        handle_weather_alert(event['data'])
    
    return jsonify({'status': 'received'}), 200

def verify_signature(payload, signature):
    expected = hmac.new(
        WEBHOOK_SECRET.encode(),
        payload,
        hashlib.sha256
    ).hexdigest()
    
    return hmac.compare_digest(expected, signature)

Monitoring Recommendations

# Example monitoring setup
import logging
from datetime import datetime

class APIMonitor:
    def __init__(self):
        self.logger = logging.getLogger('weather_api')
        self.metrics = {
            'requests': 0,
            'errors': 0,
            'latency': []
        }
    
    def log_request(self, endpoint, duration, status):
        self.metrics['requests'] += 1
        self.metrics['latency'].append(duration)
        
        if status >= 400:
            self.metrics['errors'] += 1
            self.logger.error(f"API Error: {endpoint} returned {status}")
        
        # Alert if error rate > 5%
        error_rate = self.metrics['errors'] / self.metrics['requests']
        if error_rate > 0.05:
            self.send_alert("High API error rate detected")
    
    def get_average_latency(self):
        return sum(self.metrics['latency']) / len(self.metrics['latency'])

Migration Script Example

// Adapter pattern for smooth migration
class WeatherAPIAdapter {
    constructor(oldClient, newClient) {
        this.oldClient = oldClient;
        this.newClient = newClient;
        this.migrationDate = new Date('2025-03-01');
    }
    
    async getWeather(city) {
        // Gradual migration based on date
        if (new Date() < this.migrationDate) {
            // Still use old API
            return this.oldClient.getWeather(city);
        } else {
            // Use new API with format conversion
            const newData = await this.newClient.getCurrentWeather(city);
            return this.convertToOldFormat(newData);
        }
    }
    
    convertToOldFormat(newData) {
        // Map new format to old format for compatibility
        return {
            location: newData.location.city,
            temp: newData.current.temperature,
            // ... other mappings
        };
    }
}
